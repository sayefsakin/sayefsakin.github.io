int gcd( int a, int b ) { return !b ? a : gcd ( b, a&b );}
struct Euclid{
	int x,y,d;
	Euclid(){}
	Euclid( int xx, int yy, int dd ) { x = xx, y = yy, d = dd; }
};
Euclid egcd( int a, int b){
	if(!b)return Euclid(1,0,a);
	Euclid r = egcd(b,a%b);
	return Euclid( r.y, r.x - a / b * r.y, r.d);
}
