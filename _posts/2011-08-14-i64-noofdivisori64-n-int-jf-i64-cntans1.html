i64 noOfDivisor(i64 n){
	int j,f;
	i64 cnt,ans=1;
	f = 0;
	for(j=0;prime[j]<=n && j<prlen;j++){
		cnt = 0;
		if(n%prime[j]==0){
			while(n%prime[j]==0){
				cnt++;
				n /= prime[j];
			}
			f=1;
			ans *= (cnt+1);
		}
	}
	//if(!f || n!=1)cnt++;
	if(n^1)ans <<= 1;
	return ans;
}
